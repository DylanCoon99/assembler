Instruction Set Architecture

0    00000    NOP          No operation
1    00001    LDA addr     A <- RAM[addr]
2    00010    STA addr     RAM[addr] <- A
3    00011    LDI val      A <- val
4    00100    LDI B,val    B <- val
5    00101    MOV A,B      A <- B
6    00110    MOV B,A      B <- A
7    00111    LDB addr     B <- RAM[addr]
8    01000    STB addr     RAM[addr] <- B
9    01001    ADD addr     RAM[addr] <- A + B
10   01010    SUB addr     RAM[addr] <- A - B
11   01011    JMP addr     PC <- addr
12   01100    JZ addr      PC <- addr if zero
13   01101    JNZ addr     PC <- addr if not zero
14   01110    OUT          OUT <- A
15   01111    HLT          Stop Clk

Describe what instruction does

NOP -> Do nothing

LDA addr:
	Control unit writes address to data bus
	RAM reads address from the data bus
	RAM writes to data bus
	A register reads from the bus

STA addr:
	A register writes to the data bus
	RAM reads from data bus and loads the value

LDI val:
	Control unit writes value to data bus
	A register reads from the data bus

LDI B,val:
	Control unit writes value to data bus
	B register reads from the data bus

MOV A,B:
	B register writes to the data bus
	A register reads from the data bus

MOV B,A:
	A register writes to the data bus
	B register reads from the data bus

LDB addr:
	Input addr to RAM
	RAM writes to data bus
	B register reads from the bus

STB addr:
	B register writes to the data bus
	RAM reads from data bus and loads the value

ADD addr:
	ALU writes to data bus
	RAM reads from data bus and loads the value

SUB addr:
	ALU writes to data bus
	RAM reads from data bus and loads the value
	
JMP addr:
	Control unit writes the addr to the data bus
	PC reads from the data bus

JZ addr:
	if zero flag is 1: Control unit writes the addr to the data bus

JNZ addr:
	if zero flag is 0: Control unit writes the addr to the data bus

OUT:
	A register writes to the bus
	OUT register reads from the bus
